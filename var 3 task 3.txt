
#include <iostream>
#include <bitset>
#include <cstdint>

int main()
{
	uint8_t n = 0;
	uint8_t mask1 = 192; //1100 0000 в программе
	uint8_t mask2 = 3; //0000 0011 в программе
	for (int i = 0; i < 7; i++) {
		n = mask1 | n | mask2; // объединяем "гуляющие 11"
		mask1 >>= 1; // сдвигаем 11
		mask2 <<= 1; // свдигаем 11
		std::cout << std::bitset<8>(n) << std::endl;
		n = 0; //обнуляем n для дальнейшего объединения
	}
}
