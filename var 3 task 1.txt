#include <iostream>
#include <bitset>

int main()
{
	unsigned int n;
	unsigned int mask, count, maxcount;
	count = 0;
	maxcount = 0;
	bool flag = false;
	std::cin >> std::hex >> n;
	for (unsigned int i = 0; i < sizeof(n)*8; i++) {
		mask = 1 << i;
		if ((mask & n) == 0)
			flag = true; // Нашли ноль - включаем флаг
		if (flag) {
			if (mask & n) {
				count += 1;// Считаем единицы когда флаг включен
			}
			else {
				if (count > maxcount)
					maxcount = count;
				count = 0;
			}
		}
	}

	std::cout << std::bitset<32>(n) << std::endl;
	std::cout << maxcount;
}
