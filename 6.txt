#include <iostream>
#include <bitset>

int main()
{
	unsigned int x, y, k;
	std::cout << "input int X ";
	std::cin >> x;
	std::cout << "input int Y ";
	std::cin >> y;
	std::cout << "input len mask (0 - 31) ";
	std::cin >> k;

	unsigned int mask = ~0;

	unsigned int maskx = mask << k;
	unsigned int masky = mask >> (32 - k);

	unsigned newx = x & maskx;
	unsigned newy = y & masky;
	unsigned result = newx | newy;

	std::cout << std::bitset<32>(x) << " x in bin" << std::endl;
	std::cout << std::bitset<32>(y) << " y in bin" << std::endl;
	std::cout << std::bitset<32>(result) << " New X in bin" << std::endl;

}
