#include <iostream>
#include <bitset>

int main()
{
	unsigned int n, constn;
	unsigned int mask;
	std::cin >> std::hex >> n;
	constn = n;
	for (unsigned int i = 1; i < 31; i++) {
		mask = 1 << i;
		if (1 & (n >> (i + 1)) && !(1 & (n >> i)) && (1 & (n >> (i - 1)))) {
			n = n | mask;
		}
	}

	std::cout << std::bitset<32>(constn) << std::endl;
	std::cout << std::bitset<32>(n) << std::endl;
}
