#include <iostream>

int main() 
{
	unsigned int n, x;
	std::cout << "Input int num (0 <= num <= 31): " << std::endl;
	std::cin >> n;
	x = (1 << n);
	std::cout << std::dec << x << "  -  2^n in dec" << std::endl;
	std::cout << std::oct << x << "  -  2^n in oct" << std::endl;
	std::cout << std::hex << x << "  -  2^n in hex" << std::endl;
}
